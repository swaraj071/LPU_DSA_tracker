Happy number: https://leetcode.com/problems/happy-number/
``` python
class Solution:
    def isHappy(self, n: int) -> bool:
        if n==1 or n==7 : return True
        elif n<10 : return False
        else:
            s = 0
            while(n>0):
                r = n%10
                s += r*r
                n = n//10
            return self.isHappy(s)
```
