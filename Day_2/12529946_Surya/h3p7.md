Tribonacci Series : https://leetcode.com/problems/n-th-tribonacci-number/
``` Python
class Solution:
    def tribonacci(self, n: int) -> int:
        l = [0,1,1]
        for i in range(n):
            a = l[i]
            b = l[i+1]
            c = l[i+2]
            d = a+b+c
            l.append(d)
        print(l)
        return(l[n])
```   
